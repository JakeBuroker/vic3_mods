law_ecclesiarchy = { # 교구 정치	# Laws +
	group = lawgroup_distribution_of_power
	
	icon = "gfx/interface/icons/law_icons/ecclesiarchy.dds"
	
	progressiveness = -100

	disallowing_laws = {
		law_total_separation
		law_state_atheism
	}

	on_activate = {
		if = {
			limit = {
				has_law = law_type:law_elected_bureaucrats
			}
			deactivate_law = law_type:law_elected_bureaucrats
			activate_law = law_type:law_appointed_bureaucrats
		}
		scripted_effect_parties_disappearence = yes
		if = {
			limit = {
				has_modifier = missouri_compromise
			}
			remove_modifier = missouri_compromise
		}
		
		if = {
			limit = {
				NOT = { has_law = law_type:law_no_election }
			}
			if = {
				limit = {
					has_law = law_type:law_semi_constitutional
				}
				deactivate_law = law_type:law_semi_constitutional
				activate_law = law_type:law_no_election
			}
			if = {
				limit = {
					has_law = law_type:law_non_secret_ballot
				}
				deactivate_law = law_type:law_non_secret_ballot
				activate_law = law_type:law_no_election
			}
			if = {
				limit = {
					has_law = law_type:law_secret_ballot
				}
				deactivate_law = law_type:law_secret_ballot
				activate_law = law_type:law_no_election
			}
			if = {
				limit = {
					has_law = law_type:law_voter_registration
				}
				deactivate_law = law_type:law_voter_registration
				activate_law = law_type:law_no_election
			}
		}
	}

	modifier = {
		country_legitimacy_govt_total_clout_add = 180
		country_legitimacy_govt_size_add = 2
		country_legitimacy_ideological_incoherence_mult = -0.1
		country_clergymen_pol_str_mult = 1
		country_authority_add = 150
		state_pop_pol_str_add = -4
	}

	can_enact = {
		trigger_if = {
			limit = {
				c:BIC ?= ROOT
				is_subject = yes
			}
			custom_tooltip = {
				text = BIC_not_able_to_change_distribution_of_power_tt
				has_variable = britain_granted_dop_permission
			}
		}
	}
	
	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country
				AND = {
					modifier:country_can_impose_same_lawgroup_distribution_of_power_in_power_bloc_bool = yes
					has_law = scope:law
				}
			}
			AND = { # 교구 정치 국가 또는 신정 국가만이 종속국에게 교구 정치를 강요할 수 있음
				OR = {
					has_law = scope:law
					has_law = law_type:law_theocracy
				}
				can_impose_law_default = yes
			}
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
	
	ai_impose_chance = {
		value = 0
		
		if = {
			limit = {
				OR = {
					has_law = law_type:law_ecclesiarchy
					has_law = law_type:law_theocracy
				}
				OR = {
					AND = {
						power_bloc ?= { has_identity = identity:identity_religious }
						religion = scope:target_country.religion
					}
					has_strategy = ai_strategy_reactionary_agenda
				}
				scope:target_country = { # 종속국의 법이 종주국이 부과하려는 법과 너무 간극이 큰가?
					NOR = {
						has_law = law_type:law_autocracy
						has_law = law_type:law_oligarchy
					}
				}
			}
			
			add = base_impose_law_weight
		}
	}
}

law_proletariat_dictatorship = {
	group = lawgroup_distribution_of_power
	
	icon = "gfx/interface/icons/law_icons/proletariat_dictatorship.dds"
	
	progressiveness = 150

	unlocking_technologies = {
		socialism
	}
	
	disallowing_laws = {
		law_illegal_labor_unions
	}

	on_activate = {
		scripted_effect_parties_disappearence = yes
		if = {
			limit = {
				has_modifier = missouri_compromise
			}
			remove_modifier = missouri_compromise
		}

		if = {
			limit = {
				NOT = { has_law = law_type:law_no_election }
			}
			activate_law = law_type:law_no_election
		}
	}

	modifier = {
		country_legitimacy_headofstate_add = 10
		country_legitimacy_govt_total_clout_add = 120
		country_legitimacy_govt_size_add = 1
		country_laborers_pol_str_mult = 0.4
		country_farmers_pol_str_mult = 0.4
		country_machinists_pol_str_mult = 0.4
		state_political_strength_from_wealth_mult = -0.75
		state_pop_pol_str_mult = 0.5
	}
	
	can_enact = {
		trigger_if = {
			limit = {
				c:BIC ?= ROOT
				is_subject = yes
			}
			custom_tooltip = {
				text = BIC_not_able_to_change_distribution_of_power_tt
				has_variable = britain_granted_dop_permission
			}
		}
	}

	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country
				AND = {
					modifier:country_can_impose_same_lawgroup_distribution_of_power_in_power_bloc_bool = yes
					has_law = scope:law
				}
			}
			AND = {
				OR = {
					has_law = scope:law
					has_law = law_type:law_council_republic
				}
				can_impose_law_default = yes
			}
		}
	}

	ai_enact_weight_modifier = {
		value = 0
		
		if = {
			limit = { ai_has_enact_weight_modifier_journal_entries = yes }
			add = 750
		}
	}
	
	ai_impose_chance = {
		value = 0
		
		if = {
			limit = {
				has_law = law_type:law_proletariat_dictatorship
				has_strategy = ai_strategy_egalitarian_agenda	
				scope:target_country = {
					NOT = { has_law = law_type:law_universal_suffrage }
				}
			}
			
			add = base_impose_law_weight
		}

		if = {
			limit = {
				scope:target_country = {
					c:BIC ?= ROOT
					OR = {	
						has_journal_entry = je_british_dictate_plantations
						has_journal_entry = je_british_dictate_military
						has_journal_entry = je_british_dictate_universities
					}
				}
			}
			multiply = 0
		}
	}
}
