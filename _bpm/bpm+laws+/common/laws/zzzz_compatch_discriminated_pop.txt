INJECT:law_forced_labour = { # 강제노동
	group = lawgroup_discriminated_pop

	icon = "gfx/interface/icons/law_icons/forced_labour.dds"

	progressiveness = -150

	disallowing_laws = {
		law_cultural_exclusion
		law_multicultural
	}

	on_activate = {
		if = {
			limit = {
				NOR = {
					has_law = law_type:law_slavery_banned
					has_law = law_type:law_legacy_slavery
				}
			}
			custom_tooltip = {
				text = enslave_discriminated_labors_tt
				every_scope_state = {
					limit = { is_slave_state = yes }
					enslave_low_acceptance_workers = yes
				}
			}
		}
		if = {
			limit = {
				has_law = law_type:law_slavery_banned
				OR = {
					any_interest_group = {
						is_interest_group_type = ig_landowners
						is_in_government = yes
						leader = {
							NOR = {
								has_ideology = ideology:ideology_reformer
								has_ideology = ideology:ideology_modernizer_leader
							}
						}
					}
					any_interest_group = {
						is_interest_group_type = ig_rural_folk
						is_in_government = yes
						leader = {
							NOR = {
								has_ideology = ideology:ideology_reformer
								has_ideology = ideology:ideology_anarchist
								has_ideology = ideology:ideology_sovereignist_leader
								has_ideology = ideology:ideology_utopian
							}
						}
					}
					any_interest_group = {
						is_in_government = yes
						law_stance = {
							law = law_type:law_slave_trade
							value > neutral
						}
					}
					any_scope_character = {
						is_ruler = yes
						OR = {
							AND = {
								is_interest_group_type = ig_landowners
								NOR = {
									has_ideology = ideology:ideology_reformer
									has_ideology = ideology:ideology_modernizer_leader
								}
							}
							AND = {
								is_interest_group_type = ig_rural_folk
								NOR = {
									has_ideology = ideology:ideology_reformer
									has_ideology = ideology:ideology_anarchist
									has_ideology = ideology:ideology_sovereignist_leader
									has_ideology = ideology:ideology_utopian

				text = slavery_reinstitute_tt

		custom_tooltip = {
			text = qualifications_discriminated_pop_violent_suppresion_tt
			every_scope_state = {
				limit = { is_incorporated = yes }
				enserfment_low_acceptance_workers_incorporate = yes
			}
			every_scope_state = {
				limit = { is_incorporated = no }
				enserfment_low_acceptance_workers_unincorporate = yes
			}
		}
	}

	can_enact = {
		custom_tooltip = {
			text = discriminated_pop_law_can_enact_tt
			any_scope_state = {
				is_incorporated = yes
				any_scope_pop = {
					pop_acceptance <= acceptance_status_2
					percent >= 0.05
				}
			}
		}
	}

	modifier = {
		country_infamy_decay_mult = -0.1

		country_assimilation_violent_hostility_mult = -1
		country_assimilation_cultural_erasure_mult = -1
		country_political_strength_violent_hostility_mult = -1
		country_political_strength_cultural_erasure_mult = -1

			add = 0

			if = {

						NOT = { has_law = law_type:law_violent_suppresion }

						is_interest_group_type = ig_industrialists
